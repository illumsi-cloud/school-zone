/* 유틸 함수 */
function esc(s){
  return String(s==null?'':s).replace(/[&<>"']/g, m=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]));
}
function reEsc(s){return String(s).replace(/[.*+?^${}()|[\]\\]/g,'\\$&');}
function stripBracket(s){ return String(s==null?'':s).replace(/\s*\[[^\]]*\]\s*/g,'').trim(); }
function getField(row, names){
  if(!row) return '';
  for(const n of names){
    if(row[n]!=null && String(row[n]).trim()!=='') return String(row[n]).trim();
  }
  const keys = Object.keys(row);
  for(const n of names){
    const k = keys.find(k=>k.includes(n));
    if(k && row[k]!=null && String(row[k]).trim()!=='') return String(row[k]).trim();
  }
  return '';
}
function parseSchoolName(str){
  const m = str.match(/^(.+?)\((.+?)\)$/);
  if(m) return { full:m[1].trim(), short:m[2].trim() };
  return { full:str.trim(), short:null };
}
/* 하이라이트 */
function hl(text){
  const t = String(text==null?'':text);
  if(!t) return '';
  if(!STATE.lastTokens || STATE.lastTokens.length===0) return esc(t);
  const toks = [...STATE.lastTokens].sort((a,b)=>b.length-a.length).filter(x=>x.length>0);
  let out = esc(t);
  for(const tok of toks){
    const re = new RegExp(reEsc(esc(tok)),'gi');
    out = out.replace(re, m=>`<mark class="hit">${m}</mark>`);
  }
  return out;
}
/* Levenshtein 거리 (오타 보정) */
function lev(a,b){
  const m=a.length,n=b.length;
  if(!m) return n; if(!n) return m;
  const dp=Array.from({length:m+1},()=>new Array(n+1).fill(0));
  for(let i=0;i<=m;i++) dp[i][0]=i;
  for(let j=0;j<=n;j++) dp[0][j]=j;
  for(let i=1;i<=m;i++) for(let j=1;j<=n;j++){
    dp[i][j] = a[i-1]===b[j-1] ? dp[i-1][j-1] : Math.min(dp[i-1][j-1], dp[i-1][j], dp[i][j-1])+1;
  }
  return dp[m][n];
}

function updateStatusUI(){
  const listUi = document.getElementById('uploadedList');
  listUi.innerHTML = '';
  const allTabs = [...new Set([...Object.keys(STATE.database), ...STATE.emptyTabs])].sort();

    return;
  }
  allTabs.forEach(year=>{
    const li = document.createElement('li');
    li.innerHTML = STATE.database[year]
      ? `✅ <span style="color:#0046a0"><strong>${year}학년도</strong> 데이터 탑재됨 (${STATE.database[year].length}행)</span>`

  });
}
document.getElementById('fileInput').addEventListener('change', function(e){
  const file = e.target.files[0];
  const selectedYear = document.getElementById('adminYearSelect').value;
  if(!file) return;
  if(file.name !== selectedYear + '.csv'){
    setAdminStatus(`❌ 파일명이 연도와 일치해야 합니다. 예: ${selectedYear}.csv`, false);
    e.target.value = '';
    return;
  }
  Papa.parse(file, {
    header:true, skipEmptyLines:true,
    complete: function(results){
      STATE.database[selectedYear] = results.data;
      STATE.emptyTabs = STATE.emptyTabs.filter(y=>y!==selectedYear);
      updateStatusUI();
      setAdminStatus(`✅ ${selectedYear}학년도 데이터 탑재 완료 (${results.data.length}행)`, true);
    },
    error: function(err){ setAdminStatus('CSV 파싱 실패: '+err.message, false); }
  });
  e.target.value = '';
});

}

function scoreMatch(term, item){
  const t = term.toLowerCase(), it = item.toLowerCase();
  if(!t) return 0;
  if(it === t) return 100000;
  if(it.startsWith(t)) return 50000 - it.length;
  if(it.includes(t)) return 10000 - it.length;
  if(t.length >= 2){
    const d = lev(t, it);
    if(d <= 1 && Math.abs(t.length - it.length) <= 1) return 1000 - d*100 - it.length;
  }
  return 0;
}

  return toks[toks.length-1] || '';
}

/* 검색 (가중치 스코어링 + 다중 토큰 AND) */
function performSearch(q){
  if(!STATE.database[STATE.currentYear]) return;
  const rows = STATE.database[STATE.currentYear];
  const tokens = q.trim().split(/\s+/).filter(t=>t.length>0).map(t=>t.toLowerCase());
  STATE.lastTokens = tokens;
  if(tokens.length === 0){ clearResults(); return; }

  const scored = rows.map((r, idx)=>{
    let total = 0, matchedAll = true;

    const sch   = (getField(r,['초등학교명','초등학교'])||'').replace(/\s/g,'').toLowerCase();
    const dong  = (getField(r,['법정동(읍∙면)','법정동','읍면'])||'').replace(/\s/g,'').toLowerCase();
    const mid   = (getField(r,['중학구','중학교'])||'').replace(/\s/g,'').toLowerCase();
    const apt   = (getField(r,['공동주택명','공동주택'])||'').replace(/\s/g,'').toLowerCase();
    const ri    = (getField(r,['법정리(통)','법정리'])||'').replace(/\s/g,'').toLowerCase();
    const ban   = (getField(r,['행정리(반)','행정리'])||'').replace(/\s/g,'').toLowerCase();
    const bunji = (getField(r,['구역(번지)','구역'])||'').replace(/\s/g,'').toLowerCase();
    const excl  = (getField(r,['제외 구역(번지)','제외구역(번지)','제외 구역','제외구역'])||'').replace(/\s/g,'').toLowerCase();
    const fullAddr = [
      getField(r,['법정동(읍∙면)','법정동','읍면']),
      getField(r,['법정리(통)','법정리']),
      getField(r,['행정리(반)','행정리']),
      getField(r,['구역(번지)','구역'])
    ].join('').replace(/\s/g,'').toLowerCase();

    tokens.forEach(token=>{
      let s = 0;
      const nums = token.match(/\d+/g) || [];
      const textOnly = token.replace(/[0-9통반리번지길\s]/g,'');

      const elemTok = token.replace(/초등학교$/,'초');
      const midTok  = token.replace(/중학교$/,'중').replace(/중학구$/,'중');
      const midTarget = midTok.endsWith('중') ? midTok.replace(/중$/,'중학구') : midTok;

      if(sch === elemTok || sch === elemTok+'초' || sch === elemTok+'초등학교') s += 100000;
      else if(sch.startsWith(elemTok)) s += 50000;
      else if(sch.includes(elemTok)) s += 10000;

      if(apt === token || apt === token+'아파트') s += 100000;
      else if(apt.startsWith(token)) s += 50000;
      else if(apt.includes(token)) s += 10000;

      if(mid === midTok || mid === midTok+'중' || mid === midTok+'중학교' || mid === midTok+'중학구') s += 100000;
      else if(mid.startsWith(midTok)) s += 50000;
      else if(mid.includes(midTarget)) s += 10000;

      if(dong === token || dong === token+'동' || dong === token+'리') s += 100000;
      else if(dong.startsWith(token)) s += 50000;
      else if(textOnly && dong.includes(textOnly)) s += 5000;

      // 법정리·행정리 텍스트 매칭 (시골 리 이름: 검단리, 상검단 등)
      const riUnits  = ri.split(/[\/,]/).map(x=>x.trim()).filter(Boolean);
      const banUnits = ban.split(/[\/,]/).map(x=>x.trim()).filter(Boolean);
      if(riUnits.some(x=>x===token) || banUnits.some(x=>x===token)) s += 80000;
      else if(!/^\d/.test(token) && (ri.includes(token) || ban.includes(token))) s += 8000;

      // 번지 매칭: 하이픈 포함 패턴(952-22)은 정확매칭 우선, 단순 숫자는 부분매칭
      if(/\d/.test(token)){
        if(bunji.includes(token)) s += 60000;
        else if(excl.includes(token)) s += 15000;
        else nums.forEach(n=>{ if(fullAddr.includes(n)) s += 2000; });
      }

      // 추첨학교 매핑
      if(mid.includes('추첨') && [...STATE.추첨학교Map.entries()].some(([full,short])=>
        token.includes(full.replace(/\s/g,'').toLowerCase()) ||
        (short && token.includes(short.replace(/\s/g,'').toLowerCase()))
       )){
        s += 25000;
      }

      if(s === 0) matchedAll = false;
      else total += s;
    });

    return (matchedAll && total > 0) ? { row:r, score:total, idx:idx } : null;
  }).filter(x=>x!==null).sort((a,b)=>{
    if(b.score !== a.score) return b.score - a.score;
    return a.idx - b.idx;
  });

  renderResults(scored.map(x=>x.row));
}
